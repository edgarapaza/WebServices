CREATE TABLE `login` ( `idlogin` INT NOT NULL, `idconductor` VARCHAR(20) NOT NULL, `username` VARCHAR(45) NOT NULL, `userpass` VARCHAR(255) NOT NULL, `nivel` INT NOT NULL, `estado` TINYINT NOT NULL DEFAULT 1, `fecCreation` DATETIME NOT NULL, PRIMARY KEY (`idlogin`), CONSTRAINT `fk_login_conductor` FOREIGN KEY (`idconductor`) REFERENCES `conductor` (`idconductor`) ON DELETE CASCADE ON UPDATE CASCADE);
